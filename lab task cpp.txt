#include <iostream>
using namespace std;

int main() {
    int size;

    cout << "Enter size of character array: ";
    cin >> size;

    char arr[size];

    cout << "Enter " << size << " characters:" << endl;
    for(int i = 0; i < size; i++) {
        cin >> arr[i];
    }

    cout << "You entered: ";
    for(int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
